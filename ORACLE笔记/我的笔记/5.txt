declare
--变量、记录类型等的声明
type emp_record is record(
v_sal number(8,2) := 0,
v_emp_id number(10)
);
v_emp_record emp_record;
begin
  --程序执行部分
  select salary , employee_id into v_emp_record from employees where employee_id = 123;
  dbms_output.put_line(v_emp_record.v_emp_id||',salary:'||v_emp_record.v_sal);
--exception
  --异常的处理

end;
